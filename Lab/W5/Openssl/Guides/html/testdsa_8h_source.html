216EVP_PKEY *get_dsa(int dsa_bits)
217{
218 EVP_PKEY *pkey = NULL;
219 BIGNUM *priv_key, *pub_key, *p, *q, *g;
220 EVP_PKEY_CTX *pctx;
221 testdsa dsa_t;
222 OSSL_PARAM_BLD *tmpl = NULL;
223 OSSL_PARAM *params = NULL;
224
225 switch (dsa_bits) {
226 case 512:
227 set_dsa_ptr(dsa_t, 512);
228 break;
229 case 1024:
230 set_dsa_ptr(dsa_t, 1024);
231 break;
232 case 2048:
233 set_dsa_ptr(dsa_t, 2048);
234 break;
235 default:
236 return NULL;
237 }
238
239 if ((pctx = EVP_PKEY_CTX_new_from_name(NULL, "DSA", NULL)) == NULL)
240 return NULL;
241
242 priv_key = BN_bin2bn(dsa_t.priv, dsa_t.priv_l, NULL);
243 pub_key = BN_bin2bn(dsa_t.pub, dsa_t.pub_l, NULL);
244 p = BN_bin2bn(dsa_t.p, dsa_t.p_l, NULL);
245 q = BN_bin2bn(dsa_t.q, dsa_t.q_l, NULL);
246 g = BN_bin2bn(dsa_t.g, dsa_t.g_l, NULL);
247 if (priv_key == NULL || pub_key == NULL || p == NULL || q == NULL
248 || g == NULL) {
249 goto err;
250 }
251 if ((tmpl = OSSL_PARAM_BLD_new()) == NULL
252 || !OSSL_PARAM_BLD_push_BN(tmpl, OSSL_PKEY_PARAM_FFC_P,
253 p)
254 || !OSSL_PARAM_BLD_push_BN(tmpl, OSSL_PKEY_PARAM_FFC_Q,
255 q)
256 || !OSSL_PARAM_BLD_push_BN(tmpl, OSSL_PKEY_PARAM_FFC_G,
257 g)
258 || !OSSL_PARAM_BLD_push_BN(tmpl, OSSL_PKEY_PARAM_PRIV_KEY,
259 priv_key)
260 || !OSSL_PARAM_BLD_push_BN(tmpl, OSSL_PKEY_PARAM_PUB_KEY,
261 pub_key)
262 || (params = OSSL_PARAM_BLD_to_param(tmpl)) == NULL)
263 goto err;
