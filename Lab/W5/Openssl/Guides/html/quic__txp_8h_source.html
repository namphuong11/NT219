79/*
80 * Generates a datagram by polling the various ELs to determine if they want to
81 * generate any frames, and generating a datagram which coalesces packets for
82 * any ELs which do.
83 *
84 * Returns 0 on failure (e.g. allocation error or other errors), 1 otherwise.
85 *
86 * *status is filled with status information about the generated packet.
87 * It is always filled even in case of failure. In particular, packets can be
88 * sent even if failure is later returned.
89 * See QUIC_TXP_STATUS for details.
90 */
91typedef struct quic_txp_status_st {
92 int sent_ack_eliciting; /* Was an ACK-eliciting packet sent? */
93 int sent_handshake; /* Was a Handshake packet sent? */
94 size_t sent_pkt; /* Number of packets sent (0 if nothing was sent) */
95} QUIC_TXP_STATUS;

100/*
101 * Returns a deadline after which a call to ossl_quic_tx_packetiser_generate()
102 * might succeed even if it did not previously. This may return
103 * ossl_time_infinite() if there is no such deadline currently applicable. It
104 * returns ossl_time_zero() if there is (potentially) more data to be generated
105 * immediately. The value returned is liable to change after any call to
106 * ossl_quic_tx_packetiser_generate() (or after ACKM or CC state changes). Note
107 * that ossl_quic_tx_packetiser_generate() can also start to succeed for other
108 * non-chronological reasons, such as changes to send stream buffers, etc.
