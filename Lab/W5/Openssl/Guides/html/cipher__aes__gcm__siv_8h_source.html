61/* Define GSWAP8/GSWAP4 - used for BOTH little and big endian architectures */
62static ossl_inline uint32_t GSWAP4(uint32_t n)
63{
64 return (((n & 0x000000FF) << 24)
65 | ((n & 0x0000FF00) << 8)
66 | ((n & 0x00FF0000) >> 8)
67 | ((n & 0xFF000000) >> 24));
68}
69static ossl_inline uint64_t GSWAP8(uint64_t n)
70{
71 uint64_t result;
72
73 result = GSWAP4(n & 0x0FFFFFFFF);
74 result <<= 32;
75 return result | GSWAP4(n >> 32);
76}
