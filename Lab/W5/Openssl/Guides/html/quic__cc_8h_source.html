58/*
59 * Congestion control abstract interface.
60 *
61 * This interface is broadly based on the design described in RFC 9002. However,
62 * the demarcation between the ACKM and the congestion controller does not
63 * exactly match that delineated in the RFC 9002 pseudocode. Where aspects of
64 * the demarcation involve the congestion controller accessing internal state of
65 * the ACKM, the interface has been revised where possible to provide the
66 * information needed by the congestion controller and avoid needing to give the
67 * congestion controller access to the ACKM's internal data structures.
68 *
69 * Particular changes include:
70 *
71 * - In our implementation, it is the responsibility of the ACKM to determine
72 * if a loss event constitutes persistent congestion.
73 *
74 * - In our implementation, it is the responsibility of the ACKM to determine
75 * if the ECN-CE counter has increased. The congestion controller is simply
76 * informed when an ECN-CE event occurs.
77 *
78 * All of these changes are intended to avoid having a congestion controller
79 * have to access ACKM internal state.
80 */
81#define OSSL_CC_LOST_FLAG_PERSISTENT_CONGESTION (1U << 0)

107 /*
108 * (Re)bind output (diagnostic) information.
109 *
110 * params is an array of OSSL_PARAM structures used to output values. The
111 * storage locations associated with each parameter are stored internally
112 * and updated whenever the state of the congestion controller is updated;
113 * thus, the storage locations associated with the OSSL_PARAMs passed in the
114 * call to this function must remain valid until the congestion controller
115 * is freed or those parameters are unbound. A given parameter name may be
116 * bound to only one location at a time. The params structures themselves
117 * do not need to remain allocated after this call returns.
118 *
119 * Returns 1 on success and 0 on failure.
120 */
121 int (*bind_diagnostics)(OSSL_CC_DATA *ccdata,
122 OSSL_PARAM *params);
123
124 /*
125 * Unbind diagnostic information. The parameters with the given names are
126 * unbound, cancelling the effects of a previous call to bind_diagnostic().
127 * params is an array of OSSL_PARAMs. The values of the parameters are
128 * ignored. If a parameter is already unbound, there is no effect for that
129 * parameter but other parameters are still unbound.
130 *
131 * Returns 1 on success or 0 on failure.
132 */
133 int (*unbind_diagnostics)(OSSL_CC_DATA *ccdata,
134 OSSL_PARAM *params);

136 /*
137 * Returns the amount of additional data (above and beyond the data

167 /*
168 * The On Data Lost event. See OSSL_CC_LOSS_INFO structure for details
169 * of the information to be passed.
170 *
171 * Note: When the ACKM determines that a set of multiple packets has been
172 * lost, it is useful for a congestion control algorithm to be able to
173 * process this as a single loss event rather than multiple loss events.
174 * Thus, calling this function may cause the congestion controller to defer
175 * state updates under the assumption that subsequent calls to
176 * on_data_lost() representing further lost packets in the same loss event
177 * may be forthcoming. Always call on_data_lost_finished() after one or more
178 * calls to on_data_lost().
179 */
180 int (*on_data_lost)(OSSL_CC_DATA *ccdata,
181 const OSSL_CC_LOSS_INFO *info);
182
183 /*
184 * To be called after a sequence of one or more on_data_lost() calls
185 * representing multiple packets in a single loss detection incident.
186 *
187 * Flags may be 0 or OSSL_CC_LOST_FLAG_PERSISTENT_CONGESTION.
188 */
189 int (*on_data_lost_finished)(OSSL_CC_DATA *ccdata, uint32_t flags);
190
191 /*
192 * For use when a PN space is invalidated or a packet must otherwise be
193 * 'undone' for congestion control purposes without acting as a loss signal.
194 * Only the size of the packet is needed.
195 */
196 int (*on_data_invalidated)(OSSL_CC_DATA *ccdata,
197 uint64_t num_bytes);
198
199 /*
200 * Called from the ACKM when detecting an increased ECN-CE value in an ACK
201 * frame. This indicates congestion.
202 *
203 * Note that this differs from the RFC's conceptual segregation of the loss
204 * detection and congestion controller functions, as in our implementation
205 * the ACKM is responsible for detecting increases to ECN-CE and simply
206 * tells the congestion controller when ECN-triggered congestion has
207 * occurred. This allows a slightly more efficient implementation and
208 * narrower interface between the ACKM and CC.
209 */
210 int (*on_ecn)(OSSL_CC_DATA *ccdata,
211 const OSSL_CC_ECN_INFO *info);
