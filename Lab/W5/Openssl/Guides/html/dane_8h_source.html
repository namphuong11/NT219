54/*
55 * Shared DANE context
56 */
57struct dane_ctx_st {
58 const EVP_MD **mdevp; /* mtype -> digest */
59 uint8_t *mdord; /* mtype -> preference */
60 uint8_t mdmax; /* highest supported mtype */
61 unsigned long flags; /* feature bitmask */
62};

64/*
65 * Per connection DANE state
66 */
67struct ssl_dane_st {
68 struct dane_ctx_st *dctx;
69 STACK_OF(danetls_record) *trecs;
70 STACK_OF(X509) *certs; /* DANE-TA(2) Cert(0) Full(0) certs */
71 danetls_record *mtlsa; /* Matching TLSA record */
72 X509 *mcert; /* DANE matched cert */
73 uint32_t umask; /* Usages present */
74 int mdpth; /* Depth of matched cert */
75 int pdpth; /* Depth of PKIX trust */
76 unsigned long flags; /* feature bitmask */
77};
