openssl ecparam -h
openssl ec -h

2. List ECC curves
openssl ecparam -list_curves

3. Generate the ECC key
3.1 PEM format
+Private key
openssl ecparam -genkey -name prime256v1 -inform PEM -out eccprivate.pem
openssl ecparam -genkey -name prime256v1 -inform PEM -out eccprivate1.pem -param_enc explicit
openssl ecparam -in eccprivate.pem -out eccprivate2.pem -conv_form compressed
+Public key
openssl ec -in eccprivate.pem -pubout -out ecpublickey.pem
openssl ec -in eccprivate.pem -pubout -out ecpublickey1.pem -param_enc explicit
openssl ec -in eccprivate.pem -pubout -out ecpublickey2.pem -conv_form compressed

4. Prints out the public, private key components and parameters.
Read private key
openssl ec -in eccprivate.pem -conv_form uncompressed -text -noout
openssl ec -pubin -in ecpublickey1.pem -text -noout
openssl ec -pubin -in ecpublickey2.pem -text -noout

3.2 DER format
# create DER key with openssl 
openssl ecparam -name secp384r1 -genkey -outform DER -out eccprivate.der
Full parameters
openssl ecparam -name secp384r1 -genkey -outform DER -out eccprivate1.der -param_enc explicit

# print private key information
openssl ec -in eccprivate.der -inform DER -text -noout
openssl ec -in eccprivate1.der -inform DER -text -noout
# export public key
openssl ec -in eccprivate.der -inform DER -outform DER -pubout -out eccpub.der
openssl ec -in eccprivate1.der -inform DER -outform DER -pubout -out eccpub1.der

# print public key information
openssl ec -pubin -in eccpub.der -inform DER -text -noout
openssl ec -pubin -in eccpub1.der -inform DER -text -noout
